        // Update date and time
        function updateDateTime() {
            const now = new Date();
            const dateOptions = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
            const timeOptions = { hour: '2-digit', minute: '2-digit', second: '2-digit' };
            
            document.getElementById('currentDate').textContent = now.toLocaleDateString('en-US', dateOptions);
        }

        // Update Market Data (Simulated Live Updates)
        function updateMarketData() {
            marketData.forEach(item => {
                // Randomly update values
                const changeAmount = (Math.random() * 0.4 - 0.2).toFixed(2);
                const newChange = (parseFloat(item.change.replace('%', '')) + parseFloat(changeAmount)).toFixed(1);
                item.change = (newChange > 0 ? '+' : '') + newChange + '%';
                item.positive = newChange > 0;
                
                // Update value slightly
                if (item.symbol !== 'USD/EUR' && item.symbol !== 'GBP/USD' && item.symbol !== '10Y YIELD') {
                    const valueChange = (Math.random() * 20 - 10).toFixed(2);
                    const currentValue = parseFloat(item.value.replace(/[^0-9.-]/g, ''));
                    const newValue = (currentValue + parseFloat(valueChange)).toFixed(2);
                    
                    if (item.symbol === 'GOLD' || item.symbol === 'OIL' || item.symbol === 'BITCOIN') {
                        item.value = item.symbol === 'BITCOIN' ? '$' + newValue : '$' + newValue;
                    } else {
                        item.value = newValue.includes('.') ? newValue : newValue + '.00';
                    }
                }
            });
            
            // Re-render ticker with updated data
            initMarketTicker();
        }

        // Initialize all components
        function initializeAll() {
            initMarketTicker();
            initBreakingNews();
            renderArticles('business');
            renderArticles('economic');
            renderArticles('market');
            updateDateTime();
            
            // Update market data every 5 seconds to simulate live updates
            setInterval(updateMarketData, 5000);
            
            // Update time every second
            setInterval(updateDateTime, 1000);
        }

        document.addEventListener('DOMContentLoaded', function() {

                    const target = e.target.getAttribute('href').replace('#', '');
                    if (['business', 'economic', 'market'].includes(target)) {
                        renderArticles(target);
                    }
                });
            });
        });
